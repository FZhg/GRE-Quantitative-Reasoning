			\begin{aligned}
			&\because AC = AE\\
			&\therefore\angle ACE = \angle AEC\\

	\framesubtitle{多边形内角和}
	\begin{theorem}

the sum of the measures of the interior angles of an n-sided polygon is $(n − 2)(\ang{180})$.
